<script>
function timer(endtime){
    var d,h,s,m;
    var t= Date.parse(endtime)-Date.parse(new Date());
    s=Math.floor((t/1000)%60);
    m=Math.floor((t/(1000*60))%60);
    h=Math.floor((t/(1000*60*60))%24);    
    d=Math.floor((t/(1000*60*60*24)));
    return{
    'total':t,
    'days':d,
    'hours':h,
    'minutes':m,
    'seconds':s
        };
}
function renew(id,endtime){
    var c=document.getElementById(id);
    daysspan=c.querySelector('.days');    
    hoursspan=c.querySelector('.hours');
    minutesspan=c.querySelector('.minutes');
    secondsspan=c.querySelector('.seconds');
    
    function update(){
    var t=timer(endtime);
     daysspan.innerHTML=t.days;
    hoursspan.innerHTML=('0'+t.hours).slice(-2);
    minutesspan.innerHTML=('0'+t.minutes).slice(-2);
    secondsspan.innerHTML=('0'+t.seconds).slice(-2);
    if(t.total<=0){
        clearInterval(timeinterval);
        }
           }
    update();
    var timeinterval=setInterval(update,1000);
}
var deadline = new Date(2016,10,8,10,30,0,0);//For elections on November 8,2016 at 10:30 a.m.
renew('clock', deadline);
</script>
